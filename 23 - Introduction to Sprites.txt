import java.awt.Image;

public class Sprite {
	
	private Animation a;
	private float x;
	private float y;
	private float vx;
	private float vy;
	
	//CONSTRUCTOR
	public Sprite(Animation a){
		this.a = a;
		
	}
	
	//Change position 
	public void update (long timePassed){
		x += vx * timePassed;
		y += vy * timePassed;
		a.update(timePassed);
	}
}
